# 1.2 PCD点云数据预处理（DPM：点云特征 → 四维时空偏差）
def preprocess_dpm_data(pcd_dir):
    pcd_files = [f for f in os.listdir(pcd_dir) if f.endswith(".pcd")]
    X_pcd, y_dev = [], []
    # 遍历点云文件，提取核心特征（工业点云轻量化特征）
    for file in pcd_files:
        pcd = o3d.io.read_point_cloud(os.path.join(pcd_dir, file))
        points = np.asarray(pcd.points)
        # 提取点云特征：均值/方差/最值/中心点（兼顾精度与算力）
        pcd_feat = np.concatenate([
            np.mean(points, axis=0), np.var(points, axis=0),
            np.max(points, axis=0), np.min(points, axis=0),
            np.median(points, axis=0)
        ]).flatten()
        # 从文件名解析偏差标签（格式：xxx_偏差X_偏差Y_偏差Z_偏差T.pcd）
        dev_vals = [float(x) for x in file.split("_")[1:-1]]
        if len(dev_vals) == DEVIATION_DIM:
            X_pcd.append(pcd_feat)
            y_dev.append(dev_vals)
    # 标准化处理
    X_pcd, y_dev = np.array(X_pcd), np.array(y_dev)
    scaler_X = StandardScaler()
    scaler_y = StandardScaler()
    X_scaled = scaler_X.fit_transform(X_pcd)
    y_scaled = scaler_y.fit_transform(y_dev)
    # 划分训练集/验证集并转换为Tensor
    X_train, X_val, y_train, y_val = train_test_split(X_scaled, y_scaled, test_size=0.2, random_state=42)
    return (
        torch.FloatTensor(X_train).to(DEVICE),
        torch.FloatTensor(X_val).to(DEVICE),
        torch.FloatTensor(y_train).to(DEVICE),
        torch.FloatTensor(y_val).to(DEVICE),
        scaler_X, scaler_y,
        X_pcd.shape[1]  # 返回点云特征维度，用于DPM模型初始化
    )

# ===================== 3. 模型训练核心函数（含早停+动态学习率）=====================
def train_industrial_model(model, train_loader, val_loader, criterion, optimizer, scheduler):
    best_val_loss = float('inf')
    early_stop_count = 0
    # 训练循环
    for epoch in range(EPOCHS):
        model.train()
        train_loss = 0.0
        # 进度条可视化，适配工业现场训练监控
        for X, y in tqdm(train_loader, desc=f"Epoch {epoch+1}/{EPOCHS}"):
            X, y = X.to(DEVICE), y.to(DEVICE)
            optimizer.zero_grad()
            outputs = model(X)
            loss = criterion(outputs, y)
            loss.backward()
            optimizer.step()
            train_loss += loss.item() * X.size(0)
        # 验证阶段，关闭梯度计算
        model.eval()
        val_loss = 0.0
        with torch.no_grad():
            for X, y in val_loader:
                X, y = X.to(DEVICE), y.to(DEVICE)
                outputs = model(X)
                val_loss += criterion(outputs, y).item() * X.size(0)
        # 计算平均损失
        train_loss /= len(train_loader.dataset)
        val_loss /= len(val_loader.dataset)
        # 动态调整学习率
        scheduler.step()
        # 打印训练日志
        print(f"Train Loss: {train_loss:.6f} | Val Loss: {val_loss:.6f} | LR: {optimizer.param_groups[0]['lr']:.6f}")
        # 早停机制：防止过拟合，节省工业训练时间
        if val_loss < best_val_loss:
            best_val_loss = val_loss
            torch.save(model.state_dict(), os.path.join(OUTPUT_ROOT, "temp_best_model.pth"))
            early_stop_count = 0
        else:
            early_stop_count += 1
            if early_stop_count >= PATIENCE:
                print(f"早停触发！最佳验证损失：{best_val_loss:.6f}")
                break
    # 加载最佳模型并删除临时文件
    model.load_state_dict(torch.load(os.path.join(OUTPUT_ROOT, "temp_best_model.pth")))
    os.remove(os.path.join(OUTPUT_ROOT, "temp_best_model.pth"))
    return model, best_val_loss

def prune_industrial_model(model, prune_ratio=PRUNE_RATIO):
    # 对全连接层进行结构化L1剪枝，保留核心权重
    for name, module in model.named_modules():
        if isinstance(module, nn.Linear):
            prune.l1_unstructured(module, name='weight', amount=prune_ratio)
            prune.remove(module, 'weight')  # 移除剪枝掩码，固化模型
    print(f"模型轻量化完成，剪枝比例：{prune_ratio*100}%")
    return model

    return mse, r2

# 2. PCD点云数据预处理（DPM模型：点云特征→四维偏差）
def preprocess_pcd_data(pcd_dir):
    pcd_files = [f for f in os.listdir(pcd_dir) if f.endswith(".pcd")]
    X_pcd, y_dev = [], []
    for file in pcd_files:
        # 读取点云文件
        pcd = o3d.io.read_point_cloud(os.path.join(pcd_dir, file))
        points = np.asarray(pcd.points)
        # 提取点云核心特征（工业点云常用：均值、方差、最值、中心点）
        pcd_feature = np.concatenate([
            np.mean(points, axis=0), np.var(points, axis=0),
            np.max(points, axis=0), np.min(points, axis=0),
            np.median(points, axis=0)
        ]).flatten()
        # 读取偏差标签（文件名格式：xxx_偏差x_偏差y_偏差z_偏差t.pcd）
        dev_vals = [float(x) for x in file.split("_")[1:-1]]
        if len(dev_vals) == DEVIATION_PARAM_DIM:
            X_pcd.append(pcd_feature)
            y_dev.append(dev_vals)
    # 转换为数组并标准化
    X_pcd = np.array(X_pcd)
    y_dev = np.array(y_dev)
    scaler_X_pcd = StandardScaler()
    scaler_y_dev = StandardScaler()
    X_pcd_scaled = scaler_X_pcd.fit_transform(X_pcd)
    y_dev_scaled = scaler_y_dev.fit_transform(y_dev)
    # 划分训练集/验证集并转换为Tensor
    X_train, X_val, y_train, y_val = train_test_split(X_pcd_scaled, y_dev_scaled, test_size=0.2, random_state=SEED)
    X_train = torch.FloatTensor(X_train).to(device)
    X_val = torch.FloatTensor(X_val).to(device)
    y_train = torch.FloatTensor(y_train).to(device)
    y_val = torch.FloatTensor(y_val).to(device)
    return X_train, X_val, y_train, y_val, scaler_X_pcd, scaler_y_dev, X_pcd.shape[1]

# ===================== 模型训练函数（含早停+正则化+进度条）=====================
def train_model(model, train_loader, val_loader, criterion, optimizer, scheduler, epochs, patience):
    best_val_loss = float('inf')
    early_stop_count = 0
    train_loss_list, val_loss_list = [], []
    # 训练循环
    for epoch in range(epochs):
        model.train()
        train_loss = 0.0
        # 训练进度条（工业训练可视化）
        for X, y in tqdm(train_loader, desc=f"Epoch {epoch+1}/{epochs}"):
            X, y = X.to(device), y.to(device)
            optimizer.zero_grad()
            outputs = model(X)
            loss = criterion(outputs, y)
            loss.backward()
            optimizer.step()
            train_loss += loss.item() * X.size(0)
        # 验证阶段
        model.eval()
        val_loss = 0.0
        with torch.no_grad():
            for X, y in val_loader:
                X, y = X.to(device), y.to(device)
                outputs = model(X)
                loss = criterion(outputs, y)
                val_loss += loss.item() * X.size(0)
        # 计算平均损失
        train_loss /= len(train_loader.dataset)
        val_loss /= len(val_loader.dataset)
        train_loss_list.append(train_loss)
        val_loss_list.append(val_loss)
        # 学习率调度
        scheduler.step()
        # 打印训练信息
        print(f"Train Loss: {train_loss:.6f} | Val Loss: {val_loss:.6f} | LR: {optimizer.param_groups[0]['lr']:.6f}")
        # 早停机制（工业训练防止过拟合）
        if val_loss < best_val_loss:
            best_val_loss = val_loss
            torch.save(model.state_dict(), os.path.join(OUTPUT_ROOT, "best_model.pth"))
            early_stop_count = 0
        else:
            early_stop_count += 1
            if early_stop_count >= patience:
                print(f"早停触发！最佳验证损失：{best_val_loss:.6f}")
                break
    # 加载最佳模型
    model.load_state_dict(torch.load(os.path.join(OUTPUT_ROOT, "best_model.pth")))
    os.remove(os.path.join(OUTPUT_ROOT, "best_model.pth"))  # 删除临时文件
    return model, train_loss_list, best_val_loss

# ===================== 模型轻量化剪枝（工业边缘设备适配）=====================
def prune_model(model, prune_ratio=PRUNE_RATIO):
    # 对全连接层进行结构化剪枝
    for name, module in model.named_modules():
        if isinstance(module, nn.Linear):
            prune.l1_unstructured(module, name='weight', amount=prune_ratio)
            prune.remove(module, 'weight')
    print("模型轻量化剪枝完成，剪枝比例：{}%".format(prune_ratio*100))
    return model

# ===================== 模型评估函数（工业指标：MSE/R2）=====================
def evaluate_model(model, val_loader, scaler_y):
    model.eval()
    y_true, y_pred = [], []
    with torch.no_grad():
        for X, y in val_loader:
            X, y = X.to(device), y.to(device)
            outputs = model(X)
            # 反标准化，还原为工业实际参数
            y_true.extend(scaler_y.inverse_transform(y.cpu().numpy()))
            y_pred.extend(scaler_y.inverse_transform(outputs.cpu().numpy()))
    # 计算工业评估指标：均方误差（MSE）、决定系数（R2）
    mse = mean_squared_error(y_true, y_pred)
    r2 = r2_score(y_true, y_pred)
    print(f"模型评估结果 | MSE: {mse:.6f} | R2: {r2:.6f}")
    print(f"R2≥0.95为工业合格标准，当前R2：{r2:.6f}")
    return mse, r2

# ===================== 唯一Task ID生成（公式系统专属）=====================
def generate_task_id(formula_type: str, env: str = "dev") -> str:
    """
    生成Task ID：时间戳+公式分类+环境+短uuid，如20260530_mechanics_prod_9f2e

    """

    task_id = f"{time_str}_{formula_type}_{env}_{short_uuid}"
    return task_id

# ===================== 公式系统专属：量纲计算/验证辅助函数 =====================
def calculate_dimension(formula_str: str) -> str:
    """量纲计算：基于sympy解析公式量纲，异常返回error"""
    try:
        formula = sp.sympify(formula_str)
        # 此处替换为你的量纲计算核心逻辑
        dimension = sp.srepr(formula)
        return str(dimension)
    except Exception as e:
        logger.error(f"公式量纲计算失败：{formula_str}，错误：{str(e)}")
        return "error"

def verify_dimension(dim1: str, dim2: str, threshold: float = 0.95) -> bool:
    """量纲验证：对比相似度，超过阈值返回True"""
    if dim1 == "error" or dim2 == "error":
        return False
    # 此处替换为你的量纲相似度计算核心逻辑
    similarity = 1.0 if dim1 == dim2 else 0.0
    return similarity >= threshold

def set_prod_resource_limit():
    """生产环境专属资源限制，公式系统训练场景适配"""
    # 1. 内存限制：根据生产服务器调整（示例：64G，单位字节）
    MEMORY_LIMIT = 64 * 1024 * 1024 * 1024
    resource.setrlimit(resource.RLIMIT_AS, (MEMORY_LIMIT, resource.RLIM_INFINITY))
    # 2. 指定GPU卡号：与生产服务器GPU数量匹配，避免占用所有GPU（示例：0,1号卡）
    os.environ["CUDA_VISIBLE_DEVICES"] = "0,1"
    # 3. CPU线程限制：根据服务器核数调整（示例：8线程）
    os.environ["OMP_NUM_THREADS"] = "8"
    os.environ["MKL_NUM_THREADS"] = "8"
    # 4. CUDA稳定配置：保证生产训练可复现，避免硬件波动
    torch.backends.cudnn.deterministic = True
    torch.backends.cudnn.benchmark = False
    torch.cuda.empty_cache()  # 初始化清空CUDA缓存
    # 5. 资源配置验证日志
    print(f"【生产环境】资源限制已生效 | GPU卡号：{os.environ['CUDA_VISIBLE_DEVICES']} | 内存限制：{MEMORY_LIMIT/1024**3}G | CPU线程：{os.environ['OMP_NUM_THREADS']}")

# ===================== 生产环境企业级告警（公式系统专属）=====================
import requests
import json

def send_prod_alert(alert_type: str, task_id: str, msg: str, webhook_url: str = "你的企业微信/钉钉机器人webhook"):
    """
    生产告警推送：支持任务失败/量纲阈值过低/训练完成
    :param alert_type: alert_type: success/error/warn（对应训练完成/任务失败/量纲阈值低）
    :param task_id: 训练任务ID，便于追溯
    :param msg: 告警详细信息
    :param webhook_url: 企业微信/钉钉机器人webhook（替换为你的实际地址）
    """
    alert_title = {
        "success": "【公式系统-生产训练完成】",
        "error": "【公式系统-生产训练失败】",
        "warn": "【公式系统-生产告警：量纲通过率过低】"
    }[alert_type]
    # 企业微信/钉钉通用告警格式（无需修改）
    data = {
        "msgtype": "text",
        "text": {
            "content": f"{alert_title}\nTask ID：{task_id}\n详情：{msg}"
        }
    }
    try:
        requests.post(webhook_url, data=json.dumps(data), headers={"Content-Type": "application/json"})
        logger.info(f"生产告警推送成功 | 类型：{alert_type} | Task ID：{task_id}")
    except Exception as e:

import requests

def send_prod_alert(alert_type: str, task_id: str, msg: str, webhook_url: str = "你的企业微信/钉钉机器人webhook"):
    """
    生产告警推送：支持任务失败/量纲阈值过低/训练完成
    :param alert_type: alert_type: success/error/warn（对应训练完成/任务失败/量纲阈值低）
    :param task_id: 训练任务ID，便于追溯
    :param msg: 告警详细信息

    """
    alert_title = {
        "success": "【公式系统-生产训练完成】",
        "error": "【公式系统-生产训练失败】",
        "warn": "【公式系统-生产告警：量纲通过率过低】"
